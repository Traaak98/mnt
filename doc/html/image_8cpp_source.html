326void image::find_shadow(double &shadow, double x1, double y1, double z1, double x2, double y2, double z2, double x3, double y3, double z3){

328 // CHoix du cote d'ou vient la lumiere
329 double altitude = (90-45)*M_PI/180;
330 double azimuth = (360-315+90)*M_PI/180;
331
332 // Calcul du vecteur normal au triangle
333 double n1 = (y2-y1)*(z3-z1)-(z2-z1)*(y3-y1);
334 double n2 = (z2-z1)*(x3-x1)-(x2-x1)*(z3-z1);
335 double n3 = (x2-x1)*(y3-y1)-(y2-y1)*(x3-x1);
336
337 shadow = (cos(azimuth)*cos(altitude)*n1+sin(azimuth)*cos(altitude)*n2-sin(altitude)*n3)
338 /(sqrt(pow(n1,2)+pow(n2, 2)+pow(n3,2))*sqrt(pow(cos(azimuth)*cos(altitude),2)+pow(sin(azimuth)*cos(altitude),2)+pow(-sin(altitude),2)));

341 if(shadow<0){shadow=0;}
